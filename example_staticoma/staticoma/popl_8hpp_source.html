704 template<class T>
705 inline void Value<T>::parse(OptionName what_name, const char* value)
706 {
707  T parsed_value;
708  std::string strValue;
709  if (value != nullptr)
710  strValue = value;
711 
712  std::istringstream is(strValue);
713  int valuesRead = 0;
714  while (is.good())
715  {
716  if (is.peek() != EOF)
717  is >> parsed_value;
718  else
719  break;
720 
721  valuesRead++;
722  }
723 
724  if (is.fail())
725  throw invalid_option(this, invalid_option::Error::invalid_argument, what_name, value, "invalid argument for " + name(what_name, true) + ": '" + strValue + "'");
726 
727  if (valuesRead > 1)
728  throw invalid_option(this, invalid_option::Error::too_many_arguments, what_name, value, "too many arguments for " + name(what_name, true) + ": '" + strValue + "'");
729 
730  if (strValue.empty())
731  throw invalid_option(this, invalid_option::Error::missing_argument, what_name, "", "missing argument for " + name(what_name, true));
732 
733  this->add_value(parsed_value);
734 }

919 inline void OptionParser::parse(int argc, const char * const argv[])
920 {
921  unknown_options_.clear();
922  non_option_args_.clear();
923  for (auto& opt : options_)
924  opt->clear();
925 
926  for (int n=1; n<argc; ++n)
927  {
928  const std::string arg(argv[n]);
929  if (arg == "--")
930  {
931  ///from here on only non opt args
932  for (int m=n+1; m<argc; ++m)
933  non_option_args_.emplace_back(argv[m]);
934  }
935  else if (arg.find("--") == 0)
936  {
937  /// long option arg
938  std::string opt = arg.substr(2);
939  std::string optarg;
940  size_t equalIdx = opt.find('=');
941  if (equalIdx != std::string::npos)
942  {
943  optarg = opt.substr(equalIdx + 1);
944  opt.resize(equalIdx);
945  }
946 
947  Option_ptr option = find_option(opt);
948  if (option && (option->attribute() == Attribute::inactive))
949  option = nullptr;
950  if (option)
951  {
952  if (option->argument_type() == Argument::no)
953  {
954  if (!optarg.empty())
955  option = nullptr;
956  }
957  else if (option->argument_type() == Argument::required)
958  {
959  if (optarg.empty() && n < argc-1)
960  optarg = argv[++n];
961  }
962  }
963 
964  if (option)
965  option->parse(OptionName::long_name, optarg.c_str());
966  else
967  unknown_options_.push_back(arg);
968  }
969  else if (arg.find('-') == 0)
970  {
971  /// short option arg
972  std::string opt = arg.substr(1);
973  bool unknown = false;
974  for (size_t m=0; m<opt.size(); ++m)
975  {
976  char c = opt[m];
977  std::string optarg;
978 
979  Option_ptr option = find_option(c);
980  if (option && (option->attribute() == Attribute::inactive))
981  option = nullptr;
982  if (option)
983  {
984  if (option->argument_type() == Argument::required)
985  {
986  /// use the rest of the current argument as optarg
987  optarg = opt.substr(m + 1);
988  /// or the next arg
989  if (optarg.empty() && n < argc-1)
990  optarg = argv[++n];
991  m = opt.size();
992  }
993  else if (option->argument_type() == Argument::optional)
994  {
995  /// use the rest of the current argument as optarg
996  optarg = opt.substr(m + 1);
997  m = opt.size();
998  }
999  }
1000 
1001  if (option)
1002  option->parse(OptionName::short_name, optarg.c_str());
1003  else
1004  unknown = true;
1005  }
1006  if (unknown)
1007  unknown_options_.push_back(arg);
1008  }
1009  else
1010  {
1011  non_option_args_.push_back(arg);
1012  }
1013  }
1014 
1015  for (auto& opt : options_)
1016  {
1017  if ((opt->attribute() == Attribute::required) && !opt->is_set())
1018  {
1019  std::string option = opt->long_name().empty()?std::string(1, opt->short_name()):opt->long_name();
1020  throw invalid_option(opt.get(), invalid_option::Error::missing_option, "option \"" + option + "\" is required");
1021  }
1022  }
1023 }

1077 inline std::string ConsoleOptionPrinter::print(const Attribute& max_attribute) const
1078 {
1079  if (option_parser_ == nullptr)
1080  return "";
1081 
1082  if (max_attribute < Attribute::optional)
1083  throw std::invalid_argument("attribute must be 'optional', 'advanced', or 'default'");
1084 
1085  std::stringstream s;
1086  if (!option_parser_->description().empty())
1087  s << option_parser_->description() << ":\n";
1088 
1089  size_t optionRightMargin(20);
1090  const size_t maxDescriptionLeftMargin(40);
1091 // const size_t descriptionRightMargin(80);
1092 
1093  for (const auto& option: option_parser_->options())
1094  optionRightMargin = (std::max)(optionRightMargin, to_string(option).size() + 2);
1095  optionRightMargin = (std::min)(maxDescriptionLeftMargin - 2, optionRightMargin);
1096 
1097  for (const auto& option: option_parser_->options())
1098  {
1099  if ((option->attribute() <= Attribute::hidden) ||
1100  (option->attribute() > max_attribute))
1101  continue;
1102  std::string optionStr = to_string(option);
1103  if (optionStr.size() < optionRightMargin)
1104  optionStr.resize(optionRightMargin, ' ');
1105  else
1106  optionStr += "\n" + std::string(optionRightMargin, ' ');
1107  s << optionStr;
1108 
1109  std::string line;
1110  std::vector<std::string> lines;
1111  std::stringstream description(option->description());
1112  while (std::getline(description, line, '\n'))
1113  lines.push_back(line);
1114 
1115  std::string empty(optionRightMargin, ' ');
1116  for (size_t n=0; n<lines.size(); ++n)
1117  {
1118  if (n > 0)
1119  s << "\n" << empty;
1120  s << lines[n];
1121  }
1122  s << "\n";
1123  }
1124 
1125  return s.str();
1126 }
